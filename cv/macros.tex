\newcommand{\name}[1]{
    % Prints name
    %
    % 1 - name
    %\thispagestyle{empty}
    \flushleft{
        {\huge{\bf{#1}}}
    }
}

\newcommand{\contact}[3]{
    % Prints contact info
    %
    % 1 - Address, first line
    % 3 - Email
    % 4 - Phone number
    \setstretch{1.25}
    \vspace{0.5cm}

    \begin{minipage}[t]{0.5\textwidth}
        \flushleft{
            #1\\
            #2 \quad #3 \\
            LinkedIn: \href{https://www.linkedin.com/in/alexander-cumberworth}{alexander-cumberworth}, GitHub: \href{https://github.com/cumberworth}{cumberworth}
        }%
    \end{minipage} \hfill
    \begin{minipage}[t]{0.45\textwidth}
        \setstretch{1}
        \flushleft{Nationality: Canadian (Tier 4 Doctorate Extension visa, with permission to work in the UK for 12 months and option to switch into Tier 2 visa after this)}
    \end{minipage}
    \setstretch{1}
}

\newcommand{\journal}[7]{
    % Journal article
    %
    % 1 - Author list
    % 2 - Title
    % 3 - Journal abbreviation
    % 4 - Issue number
    % 5 - Page numbers
    % 6 - Year
    % 7 - DOI
    #1. #2. \href{#7}{\textit{#3} \textbf{#4}, #5 (#6).}
}

\newcommand{\book}[7]{
    % Book chapter
    %
    % 1 - Author list
    % 2 - Chapter title
    % 3 - Book title
    % 4 - Editor
    % 5 - Publisher, short form
    % 6 - Page numbers
    % 7 - Year
    #1. #2. \textit{#3}. #4, (Ed.) #5, pp. #6 (#7).
}

\newcommand{\prep}[2]{
    % Manuscript in preparation
    %
    % 1 - Author list
    % 2 - Title
    #1. #2. In preparation.
}

\newcommand{\degree}[5]{
    % Degree
    %
    % 1 - Type
    % 2 - Title
    % 3 - Institution
    % 4 - Years
    % 5 - Description
    \textbf{#1, #2}, #3 \hfill #4 \\
    \vspace{0.15cm}
    \hspace{0.5cm}
    \begin{minipage}{0.8\textwidth}
        #5
    \end{minipage}
}

\newcommand{\research}[4]{
    % Research experience
    %
    % 1 - Position
    % 2 - Institution
    % 3 - Years
    % 4 - Description
    \textbf{#1}, #2 \hfill #3 \\
    \vspace{0.20cm}
    \begin{minipage}{0.85\textwidth}
        \begin{itemize}
            #4
        \end{itemize}
    \end{minipage}
}

\newcommand{\award}[3]{
    % Awards and honours
    %
    % 1 - Award name
    % 2 - Year
    % 3 - Presenter
    \textbf{#1} \hfill #2 \\
    #3 \vspace{0.2cm}
}

\newcommand{\skill}[2]{
    % Skill
    %
    % 1 - Skill
    % 2 - Description
    \textbf{#1} \\
    \vspace{0.15cm}
    \begin{minipage}{0.95\textwidth}
        \begin{itemize}
        #2
        \end{itemize}
    \end{minipage}%
}

\newcommand{\reference}[5]{
    % Reference
    %
    % 1 - Name
    % 2 - University
    % 3 - Address
    % 4 - Phone number
    % 5 - Email
    \textbf{#1}, #2 \\[0.1cm]
    #3 \\[0.1cm]
    #4, #5
}
